const int pirPin = 3;
int pirState = LOW;
int peopleCount = 0;
void setup() {
  pinMode(pirPin, INPUT);
  Serial.begin(9600);
  Serial.println("People Counter Initialized");
}
void loop() {
  int pirValue = digitalRead(pirPin);
  if (pirValue == HIGH)
  {
    if (pirState == LOW) 
    {
      peopleCount++;
      Serial.print("Person went in! Total count: ");
      Serial.println(peopleCount);
      delay(1000);
      pirState = HIGH;
    }
  }
  else 
  {
    if (pirState == HIGH) 
    {
      digitalWrite(ledPin, LOW);
      digitalWrite(buzzerPin, LOW);
      pirState = LOW;
    }
  }
}
